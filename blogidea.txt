import '@react-native-firebase/app';
import {
  getMessaging,
  setBackgroundMessageHandler,
} from '@react-native-firebase/messaging';
import { getApp } from '@react-native-firebase/app';
import notifee, { AndroidImportance } from '@notifee/react-native';

const messaging = getMessaging(getApp());

setBackgroundMessageHandler(messaging, async remoteMessage => {
  try {
    const notification = remoteMessage.notification;
    const data = remoteMessage.data || {};
    const title = notification?.title ?? data?.title ?? 'Notification';
    const body = notification?.body ?? data?.body ?? data?.message ?? '';

    if (Platform.OS === 'android') {
      await notifee.createChannel({
        id: 'default',
        name: 'Default Notifications',
        description: 'Default notification channel',
        importance: AndroidImportance.HIGH,
        vibration: true,
        sound: 'default',
      });
      await notifee.displayNotification({
        title,
        body,
        data,
        android: {
          channelId: 'default',
          importance: AndroidImportance.HIGH,
          // smallIcon: 'ic_notification',
          pressAction: { id: 'default' },
        },
      });
    }
  } catch (err) {
    console.warn('Background message handler error:', err);
  }
});

import {
  checkNotifications,
  requestNotifications,
  RESULTS,
} from 'react-native-permissions';
import type { NotificationOption } from 'react-native-permissions';


const NOTIFICATION_OPTIONS: NotificationOption[] = ['alert', 'badge', 'sound'];

function isNotificationGranted(status: string): boolean {
  return status === RESULTS.GRANTED || status === RESULTS.LIMITED;
}

/** Request notification permission on app load (react-native-permissions). */
export async function requestNotificationPermissionOnLoad(): Promise<void> {
  try {
    const { status } = await checkNotifications();
    if (isNotificationGranted(status)) return;
    await requestNotifications(NOTIFICATION_OPTIONS);
  } catch {
    // Permission request can fail on simulator or if user has already denied
  }
}

this function will handle the notitifcation permisions 
we will use thid function in our App.tsx as follow

 import { getMessaging, getToken } from '@react-native-firebase/messaging';
import { getApp } from '@react-native-firebase/app';
import { saveDeviceToken } from '../api';

export async function initNotifications(userId: number) {
  if (!userId) return;

  try {
    const { status } = await checkNotifications();
    console.log('status', status);
    if (!isNotificationGranted(status)) {
      const { status: newStatus } =
        await requestNotifications(NOTIFICATION_OPTIONS);
      if (!isNotificationGranted(newStatus)) return;
    }

    const app = getApp();
    const messaging = getMessaging(app);
    const fcm_token = await getToken(messaging);
    console.log('fcm_token', fcm_token);
    await saveDeviceToken({ fcm_token }); // this api call will send the token to the backend
  } catch (_error) {
    console.log('Error', _error);
  }
}

import { useEffect, useCallback } from 'react';
import { Platform } from 'react-native';

import notifee, { AndroidImportance, EventType } from '@notifee/react-native';
import {
  getInitialNotification,
  getMessaging,
  onNotificationOpenedApp,
} from '@react-native-firebase/messaging';
import { getApp } from '@react-native-firebase/app';

const messaging = getMessaging(getApp());

const PushNotificationController = () => {
  const handleNotificationPress = useCallback(
    async (data: Record<string, string> | undefined) => {
      if (!data) return;

     
    },
    [],
  );

  const createNotificationChannel = useCallback(async () => {
    if (Platform.OS === 'android') {
      await notifee.createChannel({
        id: 'default',
        name: 'Default Notifications',
        description: 'Default notification channel',
        importance: AndroidImportance.HIGH,
        vibration: true,
        sound: 'default',
      });
      console.log('Notifee notification channel created');
    }
  }, []);

  const displayNotification = useCallback(
    async (title: string, body: string, data: any) => {
      await notifee.displayNotification({
        title,
        body,
        data,
        android: {
          channelId: 'default',
          importance: AndroidImportance.HIGH,
          // smallIcon: 'ic_notification',
          largeIcon: 'ic_launcher',
          pressAction: {
            id: 'default',
          },
        },
        ios: {
          foregroundPresentationOptions: {
            alert: true,
            badge: true,
            sound: true,
          },
        },
      });
    },
    [],
  );

  useEffect(() => {
    createNotificationChannel();

    // Handle notification opened from background
    const unsubscribeOnNotificationOpenedApp = onNotificationOpenedApp(
      messaging,
      remoteMessage => {
        console.log(
          'Notification caused app to open from background state:',
          JSON.stringify(remoteMessage),
        );
        handleNotificationPress(remoteMessage.data as Record<string, string>);
      },
    );

    // Handle notification opened from quit state
    getInitialNotification(messaging).then(remoteMessage => {
      if (remoteMessage) {
        console.log(
          'Notification caused app to open from quit state:',
          JSON.stringify(remoteMessage),
        );
        handleNotificationPress(remoteMessage.data as Record<string, string>);
      }
    });

    // Handle Notifee notification events (when user interacts with notification)
    const unsubscribeNotifeeEvents = notifee.onForegroundEvent(
      ({ type, detail }) => {
        if (type === EventType.PRESS) {
          console.log('User pressed notification:', detail.notification);
          handleNotificationPress(
            detail.notification?.data as Record<string, string>,
          );
        }
      },
    );

    // Handle background events
    notifee.onBackgroundEvent(async ({ type, detail }) => {
      if (type === EventType.PRESS) {
        console.log(
          'User pressed notification in background:',
          detail.notification,
        );
        // Background handling will be processed when app comes to foreground
      }
    });

    return () => {
      unsubscribeOnNotificationOpenedApp();
      unsubscribeNotifeeEvents();
    };
  }, [createNotificationChannel, displayNotification, handleNotificationPress]);
  return null;
};
export default PushNotificationController;

now we will handle the in app notification show
for this we have to maintain a global store which will listen for any incoming notification and push the notification details if recieved
also we need a UI component which will be consuming this store and will be responsible to show the ui accordingly

so create a global store as follow(we are using zustand you can use any other global storage library if want)
create a new file under root>src>store>useInAppNotificationStore.ts and add the following code in it
import { create } from 'zustand';

export interface InAppNotificationPayload {
  title: string;
  body?: string;
  imageUrl?: string | null;
  data?: Record<string, string>;
}

interface State {
  visible: boolean;
  title: string;
  body: string;
  imageUrl: string | null;
  data: Record<string, string> | null;
}

interface Actions {
  show: (payload: InAppNotificationPayload) => void;
  dismiss: () => void;
}

const initialState: State = {
  visible: false,
  title: '',
  body: '',
  imageUrl: null,
  data: null,
};

const useInAppNotificationStore = create<State & Actions>(set => ({
  ...initialState,
  show: payload =>
    set({
      visible: true,
      title: payload.title ?? 'Notification',
      body: payload.body ?? '',
      imageUrl: payload.imageUrl ?? null,
      data: payload.data ?? null,
    }),
  dismiss: () => set(initialState),
}));

export default useInAppNotificationStore;

now we have to create a hook which will update the store when a new notification comes
create a new hook under root>src>services>notificatios>useNotification.ts
import { useEffect } from 'react';
import { getApp } from '@react-native-firebase/app';
import { getMessaging, onMessage } from '@react-native-firebase/messaging';
import { useInAppNotificationStore } from '../../store';

export function useNotifications() {
  useEffect(() => {
    const app = getApp();
    const messaging = getMessaging(app);

    const unsubscribe = onMessage(messaging, async remoteMessage => {
      console.log('remoteMessage', remoteMessage);
      const notification = remoteMessage.notification;
      const title = notification?.title ?? 'Notification';
      const body = notification?.body ?? '';
      const imageUrl =
        (notification as { imageUrl?: string })?.imageUrl ??
        (notification as { image?: string })?.image ??
        null;
      const data = remoteMessage.data as Record<string, string> | undefined;

      useInAppNotificationStore.getState().show({
        title,
        body,
        imageUrl: imageUrl ?? undefined,
        data,
      });
    });

    return unsubscribe;
  }, []);
}
